15   public class MainActivity extends Activity { 
16    
17       private static final int DISCOVERABLE_REQUEST_CODE = 0x1; 
18       private boolean CONTINUE_READ_WRITE = true; 
19    
20       @Override 
21       protected void onCreate(Bundle savedInstanceState) { 
22           super.onCreate(savedInstanceState); 
23           setContentView(R.layout.activity_main); 
24           //Always make sure that Bluetooth server is discoverable during listening... 
25           Intent discoverableIntent = new Intent(BluetoothAdapter.ACTION_REQUEST_DISCOVERABLE); 
26           startActivityForResult(discoverableIntent, DISCOVERABLE_REQUEST_CODE); 
27       } 
28    
29       @Override 
30       protected void onActivityResult(int requestCode, int resultCode, Intent data) { 
31           super.onActivityResult(requestCode, resultCode, data); 
32           android.util.Log.e("TrackingFlow", "Creating thread to start listening..."); 
33           new Thread(reader).start(); 
34       } 
35    
36       @Override 
37       protected void onDestroy() { 
38           super.onDestroy(); 
39           if(socket != null){ 
40               try{ 
41                   is.close(); 
42                   os.close(); 
43                   socket.close(); 
44               }catch(Exception e){} 
45               CONTINUE_READ_WRITE = false; 
46           } 
47       } 
48    
49       private BluetoothSocket socket; 
50       private InputStream is; 
51       private OutputStreamWriter os; 
52       private Runnable reader = new Runnable() { 
53           public void run() { 
54               BluetoothAdapter adapter = BluetoothAdapter.getDefaultAdapter(); 
55               UUID uuid = UUID.fromString("4e5d48e0-75df-11e3-981f-0800200c9a66"); 
56               try { 
57                   BluetoothServerSocket serverSocket = adapter.listenUsingRfcommWithServiceRecord("BLTServer", uuid); 
58                   android.util.Log.e("TrackingFlow", "Listening..."); 
59                   socket = serverSocket.accept(); 
60                   android.util.Log.e("TrackingFlow", "Socket accepted..."); 
61                   is = socket.getInputStream(); 
62                   os = new OutputStreamWriter(socket.getOutputStream()); 
63                   new Thread(writter).start(); 
64    
65                   int bufferSize = 1024; 
66                   int bytesRead = -1; 
67                   byte[] buffer = new byte[bufferSize]; 
68                   //Keep reading the messages while connection is open... 
69                   while(CONTINUE_READ_WRITE){ 
70                       final StringBuilder sb = new StringBuilder(); 
71                       bytesRead = is.read(buffer); 
72                       if (bytesRead != -1) { 
73                           String result = ""; 
74                           while ((bytesRead == bufferSize) && (buffer[bufferSize-1] != 0)){ 
75                               result = result + new String(buffer, 0, bytesRead - 1); 
76                               bytesRead = is.read(buffer); 
77                           } 
78                           result = result + new String(buffer, 0, bytesRead - 1); 
79                           sb.append(result); 
80                       } 
81                       android.util.Log.e("TrackingFlow", "Read: " + sb.toString()); 
82                       //Show message on UIThread 
83                       runOnUiThread(new Runnable() { 
84                           @Override 
85                           public void run() { 
86                               Toast.makeText(MainActivity.this, sb.toString(), Toast.LENGTH_LONG).show(); 
87                           } 
88                       }); 
89                   } 
90               } catch (IOException e) {e.printStackTrace();} 
91           } 
92       }; 
93    
94       private Runnable writter = new Runnable() { 
95    
96           @Override 
97           public void run() { 
98               int index = 0; 
99               while(CONTINUE_READ_WRITE){ 
100                  try { 
101                      os.write("Message From Server" + (index++) + "\n"); 
102                      os.flush(); 
103                      Thread.sleep(2000); 
104                  } catch (Exception e) { 
105                      e.printStackTrace(); 
106                  } 
107              } 
108          } 
109      };
